server {
    server_name mccutch.org;

    listen              443 ssl;
    ssl_certificate     /etc/nginx/ssl/mccutch.org.crt;
    ssl_certificate_key /etc/nginx/ssl/mccutch.org.key;

    root /var/www/mccutch.org;
    index index.html;

    location ~ /\.(ht|git) {
        deny all;
    }
}

# redirect HTTP traffic to HTTPS
server {
    server_name         mccutch.org
    listen              80;
    return              301 https://mccutch.org$request_uri;
}
